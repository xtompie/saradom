Produce#

Nothing here is new. Visible, Filter, Val, and a small script on load. Each was shown on its own. It is composed from pieces that already exist.

    <div produce-space filter-space visible-space visible-state="category">
    <div>
        <input filter-query type="text" oninput="Filter.Filter(this)">
        <button onclick="Visible.Toggle(this, 'category', ['hidden'], ['category'])">
            <span visible-tag="category">Hide category</span>
            <span visible-tag="hidden" style="display:none">Show category</span>
        </button>
    </div>
    <ul produce-list val-tpl="[produce-row-tpl]"></ul>
    <template produce-row-tpl>
        <li filter-item>
            <i val val-fx="Pattr" val-key="match" val-attr="filter-item" hidden></i>
            <span val val-fx="Text" val-key="name"></span>
            <span val val-fx="Text" val-key="category" visible-tag="category"></span>
        </li>
    </template>
    <script>
        Produce.Init(document.currentScript, [
            { name: 'Apple',  category: 'Fruit' },
            { name: 'Banana', category: 'Fruit' },
            { name: 'Cherry', category: 'Fruit' },
            { name: 'Carrot', category: 'Vegetable' },
            { name: 'Potato', category: 'Vegetable' },
        ]);
    </script>
</div>
    const Produce = (() => {
    const Init = (ctx, items) => {
        const space = ctx.up('[produce-space]');
        space.one('[produce-list]')
            .vappend(items.map(p => ({ ...p, match: `${p.name} ${p.category}` })), space.one('[produce-row-tpl]'));
    };

    return { Init };
})();
